-- 使用SYSDBA登陆
--创建表空间（可以用管理工具创建）
CREATE TABLESPACE "XXXXXX" DATAFILE 'XXXXXXX.DBF' SIZE 1024 AUTOEXTEND ON NEXT 128 CACHE = NORMAL;
--创建用户
CREATE USER XXXXXX;
--为用户授予DBA权限
GRANT DBA TO XXXXXX;

-- 用SYSSSO登陆执行
--为用户设置初始密码
ALTER USER XXXXXX IDENTIFIED BY "TEST123456";
--为用户设置默认表空间
ALTER USER XXXXXX DEFAULT TABLESPACE "XXXXXX" DEFAULT INDEX TABLESPACE "XXXXXX";
--设置用户口令过期策略为不过期
ALTER USER XXXXXX LIMIT PASSWORD_LIFE_TIME UNLIMITED,PASSWORD_LOCK_TIME UNLIMITED,PASSWORD_GRACE_TIME UNLIMITED,FAILED_LOGIN_ATTEMPS UNLIMITED,PASSWORD_REUSE_TIME UNLIMITED,PASSWORD_REUSE_MAX UNLIMITED;
